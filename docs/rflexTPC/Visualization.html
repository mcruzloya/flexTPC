def flexTPC(T, Tmin=10, Tmax=35, rmax=1.0, α=0.8, β̃=0.2):
    result = np.zeros(len(T))
    Tidx = (T > Tmin) & (T < Tmax)
    s = α * (1 - α) / β̃**2
    result[Tidx] = rmax * np.exp(s * (α * np.log((T[Tidx] - Tmin) / α) +
                                      (1 - α) * np.log((Tmax - T[Tidx]) / (1 - α)) - 
                                      np.log(Tmax - Tmin)))
    return result

def rflexTPC(T, Tmin=10, Tmax=35, rmax=1.0, α=0.8):
    β̃ = 1 / np.sqrt(8) - (1 / np.sqrt(8) - 0.2) * np.abs(α - 0.5) / 0.3
    result = np.zeros(len(T))
    Tidx = (T > Tmin) & (T < Tmax)
    s = α * (1 - α) / β̃**2
    result[Tidx] = rmax * np.exp(s * (α * np.log((T[Tidx] - Tmin) / α) +
                                      (1.0 - α) * np.log((Tmax - T[Tidx]) / (1.0 - α)) - 
                                      np.log(Tmax - Tmin)))
    return result

def rflexTPC_nd(τ, rmax=1.0, α=0.8):
    β̃ = 1 / np.sqrt(8) - (1 / np.sqrt(8) - 0.2) * np.abs(α - 0.5) / 0.3
    result = np.zeros(len(T))
    s = α * (1 - α) / β̃**2
    result = rmax * np.exp(s * (α * np.log(τ / α) +
                                      (1.0 - α) * np.log((1.0 - τ) / (1.0 - α))))
    return result
